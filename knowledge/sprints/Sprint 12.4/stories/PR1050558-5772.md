## Description

**Tester:** Heather Clavier

**Description:** 

We are missing CustomerSignedDate from Legacy system to the new system for locums providers 

Sample:

SELECT

    c.CustomerSignedDate , a.Global_Account_ID__c, ct.Global_Contact_ID__c, cs.name

FROM UnifiedPlatformData.COREHC.HealthcarePractitionerFacility f JOIN

    UnifiedPlatformData.COREHC.CodeSet cs ON f.Facility_Title__c = [cs.Id|http://cs.Id] JOIN

    UnifiedPlatformData.COREHC.ProviderNetworkContract pnc with(nolock) ON pnc.Healthcare_Practitioner_Facility__c = [f.id|http://f.id] JOIN

    UnifiedPlatformData.COREHC.Contract c with(nolock) ON [c.id|http://c.id] = pnc.ContractId

    JOIN UnifiedPlatformData.COREHC.Contact ct ON [ct.id|http://ct.id] = f.PractitionerId JOIN

    UnifiedPlatformData.COREHC.Account a ON [a.Id|http://a.Id] = f.AccountId

WHERE ct.Global_Contact_ID__c = '814438' AND a.Global_Account_ID__c = '5813'

ORDER BY c.CustomerSignedDate desc

SELECT Agreement_Signed_Date__c, pfa.Global_Provider_ID__c, pfa.Global_Facility_ID__c, pfa.Facility_Title__c

FROM Salesforce_Live.dbo.Provider_Facility_Associations__c pfa

WHERE Global_Provider_ID__c = '814438' AND Global_Facility_ID__c = '5813'

ORDER BY pfa.Agreement_Signed_Date__c desc

Aggregate numbers:

SELECT c.CustomerSignedDate , COUNT(1)

FROM UnifiedPlatformData.COREHC.HealthcarePractitionerFacility f JOIN

    UnifiedPlatformData.COREHC.CodeSet cs ON f.Facility_Title__c = [cs.Id|http://cs.Id] JOIN

    UnifiedPlatformData.COREHC.ProviderNetworkContract pnc WITH(NOLOCK) ON pnc.Healthcare_Practitioner_Facility__c = [f.id|http://f.id] JOIN

    UnifiedPlatformData.COREHC.Contract c WITH(NOLOCK) ON [c.id|http://c.id] = pnc.ContractId

    JOIN UnifiedPlatformData.COREHC.Contact ct ON [ct.id|http://ct.id] = f.PractitionerId JOIN

    UnifiedPlatformData.COREHC.Account a ON [a.Id|http://a.Id] = f.AccountId

WHERE cs.Name ='External Locums'

GROUP BY c.CustomerSignedDate

ORDER BY c.CustomerSignedDate DESC

GO 

SELECT Agreement_Signed_Date__c, COUNT(1)

FROM Salesforce_Live.dbo.Provider_Facility_Associations__c pfa

WHERE pfa.Facility_Title__c = 'External Locums'

GROUP BY pfa.Agreement_Signed_Date__c

ORDER BY pfa.Agreement_Signed_Date__c desc
