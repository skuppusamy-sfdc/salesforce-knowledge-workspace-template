**Description:** 

I'm finding multiple HPFs with the same either EffectiveFrom or EffectiveTo date

SELECT  a.Global_Facility_ID__c, a.Name, c.Global_Provider_ID__c, MIN(hpf.Status__c) MinStatus,  MAX(hpf.Status__c)MaxStatus, MIN(hpf.EffectiveFrom)MinEffFrom, MAX(hpf.EffectiveFrom) MaxEffFrom, hpf.EffectiveTo,  c.Name, COUNT(1)

FROM UnifiedPlatformData.COREHC.HealthcarePractitionerFacility hpf JOIN

	UnifiedPlatformData.COREHC.Account a with(nolock) ON [a.id|http://a.id] = hpf.AccountId JOIN

    UnifiedPlatformData.COREHC.Contact c with(nolock) ON [c.id|http://c.id] = hpf.PractitionerId

GROUP BY a.Global_Facility_ID__c, a.Name, c.Global_Provider_ID__c, c.Name, hpf.EffectiveTo

HAVING COUNT(1) > 1

 ORDER BY  c.name, a.name

 SELECT  a.Global_Facility_ID__c, a.Name, c.Global_Provider_ID__c, MIN(hpf.Status__c) MinStatus,  MAX(hpf.Status__c)MaxStatus, hpf.EffectiveFrom, MIN(hpf.EffectiveTo) MinEffTo, MAX(hpf.EffectiveTo) MaxEffTo, c.Name, COUNT(1)

FROM UnifiedPlatformData.COREHC.HealthcarePractitionerFacility hpf JOIN

	UnifiedPlatformData.COREHC.Account a with(nolock) ON [a.id|http://a.id] = hpf.AccountId JOIN

    UnifiedPlatformData.COREHC.Contact c with(nolock) ON [c.id|http://c.id] = hpf.PractitionerId

GROUP BY a.Global_Facility_ID__c, a.Name, c.Global_Provider_ID__c, c.Name, hpf.EffectiveFrom

HAVING COUNT(1) > 1

 ORDER BY  c.name, a.name
